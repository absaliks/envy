package absaliks.envy.agent.resolvers

import absaliks.tm.commons.envy.envs.Env
import absaliks.tm.commons.envy.shell.KubernetesService
import java.util.regex.Pattern

object SecretsResolver {

  private val secretPattern = Pattern.compile("\\$\\{secret:([\\w-]+):([\\w-]+)}")

  fun resolve(env: Env, properties: Map<String, String>): Map<String, String> {
    val cache = mutableMapOf<String, String>()

    return properties
      .mapValues { (property, propertyValue) ->
        val matcher = secretPattern.matcher(propertyValue)
        if (matcher.find()) {
          KubernetesService.setContext(env.kubernetesContext!!)
          val secretName = matcher.group(1)!!
          val secretKey = matcher.group(2)!!
          val secretValue = cache["$secretName:$secretKey"]
          if (secretValue == null) {
            val entries = KubernetesService.getSecretEntries(secretName)
            entries.forEach { (key, value) ->
              cache["$secretName:$key"] = value
            }
            entries[secretKey] ?: throw IllegalStateException("Secret '$secretName' doesn't contain '$secretKey' key")
          } else {
            secretValue
          }
        } else {
          propertyValue
        }
      }
  }
}
